Detailed Description

The master transaction class extends from the AXI transaction base class svt_axi_transaction. The master transaction class contains the constraints for master specific members in the base transaction class. At the end of each transaction, the master VIP component provides object of type svt_axi_master_transaction from its analysis ports, in active and passive mode.

        }
       } else { // address constraints for coherent shareable transactions
         if (port_cfg.innershareable_start_addr.size() > 0 && port_cfg.outershareable_start_addr.size() > 0) {
            foreach (port_cfg.innershareable_start_addr[i]) {
              foreach (port_cfg.outershareable_start_addr[j]) {
                (innershareable_addr_range_index == i && outershareable_addr_range_index == j) ->
                  addr inside {[port_cfg.innershareable_start_addr[i]: port_cfg.innershareable_end_addr[i]],
                               [port_cfg.outershareable_start_addr[j]: port_cfg.outershareable_end_addr[j]]};
              }
            }
            domain_type inside {INNERSHAREABLE, OUTERSHAREABLE};
        }
        else if(port_cfg.innershareable_start_addr.size() > 0) {
            foreach (port_cfg.innershareable_start_addr[i]) {
                (innershareable_addr_range_index == i) ->
                  addr inside {[port_cfg.innershareable_start_addr[i]: port_cfg.innershareable_end_addr[i]]};
            }
            domain_type == INNERSHAREABLE;
        }
        else if(port_cfg.outershareable_start_addr.size() > 0) {
            foreach (port_cfg.outershareable_start_addr[i]) {
                (outershareable_addr_range_index == i) ->
                  addr inside {[port_cfg.outershareable_start_addr[i]: port_cfg.outershareable_end_addr[i]]};
            }
            domain_type == OUTERSHAREABLE;
        }
      } // coherent shareable transactions
    } // DomanBasedAddrGen & COHERENT & !(DVM, BARRIER)
