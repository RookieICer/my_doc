Detailed Description

The master snoop transaction class extends from the snoop transaction base class svt_axi_snoop_transaction. The master snoop transaction class contains the constraints for master specific members in the base transaction class. At the end of each transaction, the master VIP component provides object of type svt_axi_master_snoop_transaction from its analysis ports, in active and passive mode.

constraint master_snoop_transaction_delay_valid_ranges {
    cdvalid_delay.size() == snoop_burst_length;
     foreach (cdvalid_delay[index]) {
      cdvalid_delay[index] inside {[0: 16]};
    }
    acready_delay inside {[0: 16]};
    crvalid_delay inside {[0: 16]};
    if (data_before_resp) {
      reference_event_for_first_cdvalid_delay == SNOOP_ADDR_HANDSHAKE;
    }
    reference_event_for_crvalid_delay == SNOOP_ADDR_HANDSHAKE;
   }

 svt_axi_master_snoop_transaction::master_snoop_transaction_resp_valid_ranges

constraint master_snoop_transaction_resp_valid_ranges {
     solve snoop_resp_passdirty before snoop_resp_datatransfer;
    solve snoop_resp_datatransfer before snoop_data;
     if(snoop_xact_type == READUNIQUE ||
        snoop_xact_type == CLEANINVALID ||
        snoop_xact_type == MAKEINVALID) {
      snoop_resp_isshared == 0;
    }
   if(!port_cfg.data_transfer_for_makeinvalid_snoop_enable){
    if (snoop_xact_type == MAKEINVALID) {
      snoop_resp_datatransfer == 0;
      snoop_resp_passdirty == 0;
    }
   }
    if(port_cfg.ace_version == svt_axi_port_configuration::ACE_VERSION_2_0 && port_cfg.trace_tag_enable ==1) {
     if(port_cfg.loopback_trace_tag_enable){
        snoop_data_trace_tag == trace_tag;
        snoop_resp_trace_tag == trace_tag;
     }
     else {
       if(trace_tag ==1) {
         snoop_data_trace_tag ==1;
         snoop_resp_trace_tag ==1;
       }
       else {
         snoop_data_trace_tag inside {0,1};
         snoop_resp_trace_tag inside {0,1};
       }
     }
   }
    if (
         (snoop_xact_type != DVMCOMPLETE) &&
         (snoop_xact_type != DVMMESSAGE) &&
         (snoop_xact_type != MAKEINVALID) &&
         (
           (snoop_initial_cache_line_state == UNIQUEDIRTY) ||
           (snoop_initial_cache_line_state == SHAREDDIRTY)
         )
       ) {
      snoop_resp_datatransfer == 1;
    }
     snoop_resp_passdirty -> snoop_resp_datatransfer;
      if (snoop_initial_cache_line_state == INVALID) {
      snoop_resp_passdirty == 0;
      snoop_resp_isshared == 0;
      snoop_resp_datatransfer == 0;
    }
     if (
        (snoop_xact_type == READONCE) ||
        (snoop_xact_type == READCLEAN) ||
        (snoop_xact_type == READNOTSHAREDDIRTY) ||
        (snoop_xact_type == READSHARED)
       ) {
      if ((snoop_initial_cache_line_state == UNIQUECLEAN) || (snoop_initial_cache_line_state == SHAREDCLEAN)){
        snoop_resp_passdirty == 0;
      }
      if ((snoop_initial_cache_line_state == UNIQUEDIRTY) || (snoop_initial_cache_line_state == SHAREDDIRTY)) {
        !snoop_resp_isshared -> snoop_resp_passdirty;
      }
    }
    if (
        (snoop_xact_type == READUNIQUE) ||
        (snoop_xact_type == CLEANINVALID) ||
        (snoop_xact_type == MAKEINVALID)
       ) {
      snoop_resp_isshared == 0;
      if ((snoop_initial_cache_line_state == UNIQUECLEAN) || (snoop_initial_cache_line_state == SHAREDCLEAN)){
        snoop_resp_passdirty == 0;
      }
    }
    if ((snoop_xact_type == READUNIQUE) ||
        (snoop_xact_type == CLEANINVALID)
       ) {
      if ((snoop_initial_cache_line_state == UNIQUEDIRTY) || (snoop_initial_cache_line_state == SHAREDDIRTY)) {
        snoop_resp_passdirty == 1;
      }
    }
    if (snoop_xact_type == CLEANSHARED) {
      if ((snoop_initial_cache_line_state == UNIQUECLEAN) || (snoop_initial_cache_line_state == SHAREDCLEAN)){
        snoop_resp_passdirty == 0;
      }
      if ((snoop_initial_cache_line_state == UNIQUEDIRTY) || (snoop_initial_cache_line_state == SHAREDDIRTY)) {
        snoop_resp_passdirty == 1;
      }
    }
     if (snoop_xact_type == DVMCOMPLETE || snoop_xact_type == DVMMESSAGE) {
      snoop_resp_datatransfer == 0;
      snoop_resp_passdirty == 0;
      snoop_resp_isshared == 0;
      snoop_resp_wasunique == 0;
    }
    else if (((snoop_initial_cache_line_state == UNIQUECLEAN) || (snoop_initial_cache_line_state == UNIQUEDIRTY)) && snoop_resp_wasunique_enable == 1)
      snoop_resp_wasunique == 1;
    else
      snoop_resp_wasunique == 0;
    }

constraint reasonable_data {
     if (snoop_resp_datatransfer) {
       snoop_data.size() == snoop_burst_length;
       foreach (snoop_data[index]) {
         snoop_data[index] <= ((1 << port_cfg.snoop_data_width) - 1);
       }
     }
     else {
        snoop_data.size() == 0;
     }
   }
